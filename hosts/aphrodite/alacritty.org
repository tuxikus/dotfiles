Alacritty is my main terminal emulator.

* Main configuration file
The main configuration file is located at =~/.config/alacritty/alacritty.toml=.

#+begin_src toml :tangle ~/.config/alacritty/alacritty.toml :noweb yes
<<window-conf>>
<<font-conf>>
<<env-conf>>
<<catppuccin-latte>>
#+end_src

* Window configuration

#+name: window-conf
#+begin_src toml
[window]
padding = { x = 0, y = 0 }
option_as_alt = "Both"
#+end_src

* Font configuration

#+name: font-conf
#+begin_src toml
[font]
normal = { family = "UbuntuMono Nerd Font Mono", style = "Regular" }
size = 22.0
#+end_src

* Environment variables
use fish as default shell

#+name: env-conf
#+begin_src toml
[env]
SHELL = "/opt/homebrew/bin/fish"
#+end_src
